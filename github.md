# 将暂存区（Staging Area）的更改提交到本地仓库，并附带一条简短的提交信息（如这里的start settings）

git commit -m "start settings"

# 创建userB

git branch userB

# Switched to branch 'userB'

git checkout userB

# 本地新增文件

git add .文件名

# 向本地上传userB

git pull origin userB

# 向github上传userB

git push origin userB

# 查看进度

git status
